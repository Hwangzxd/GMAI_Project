tree("Root")
    fallback
        sequence
            IsHealthLessThan(0.1)
            Die
        tree("Alive")

tree("Alive")
    parallel
        repeat
            fallback
                sequence
                    IsVisible_Player
                    tree("ChaseAndAttack")
                tree("RoamAndIdle")
        repeat
            Wait(0.1) // Slight delay to allow other tasks to run

tree("ChaseAndAttack")
    fallback
        sequence
            InAttackRange
            Attack
        sequence
            IsVisible_Player
            SetDestination_Player

tree("RoamAndIdle")
    repeat
        fallback
            sequence
                IsDestinationReached
                random
                    Wait(2.0) // Idle for 2 seconds
                    FindRandomPoint
            IsVisible_Player
