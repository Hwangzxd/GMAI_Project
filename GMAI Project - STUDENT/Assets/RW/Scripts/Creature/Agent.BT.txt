tree("Root")
	parallel
		repeat mute tree("Alive")
		repeat mute tree("Die")

tree("Alive")
	repeat
		fallback
			tree("ChasePlayer")
			random
				tree("Roaming")
				tree("Idle")

tree("Roaming")
	while not IsVisible_Player
		repeat
			sequence
				IsDestinationReached
				FindRandomPoint

tree("Attack")
	sequence
		InAttackRange
		Attack

tree("ChasePlayer")
	fallback
		sequence
			InAggroRange
			tree("Attack")
		repeat 1
			sequence
				IsVisible_Player
				SetDestination_Player

tree("Idle")
	while not IsVisible_Player
		sequence
			Wait(2.0)

tree("Die")
	sequence
		IsHealthLessThan(0.1)
		Die
